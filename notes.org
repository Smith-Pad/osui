import os
import subprocess

def volume_up():
    os.system('pactl set-sink-volume @DEFAULT_SINK@ +10%')

def volume_down():
    os.system('pactl set-sink-volume @DEFAULT_SINK@ -10%')

def volume_mute():
    sound_output = subprocess.check_output(['amixer', 'get', 'Master'])
    if b'[off]' in sound_output:
        os.system('amixer sset Master unmute')
    else:
        os.system('amixer sset Master mute')
